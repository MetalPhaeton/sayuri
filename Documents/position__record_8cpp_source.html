164  void PositionRecord::ScanMember(const PositionRecord& record) {
165  // サイド毎のコピー。
166  for (Side side = 0; side < NUM_SIDES; side++) {
167  for (Piece piece_type = 0; piece_type < NUM_PIECE_TYPES; piece_type++) {
168  position_[side][piece_type] = record.position_[side][piece_type];
169  }
170  side_pieces_[side] = record.side_pieces_[side];
171  king_[side] = record.king_[side];
172  has_castled_[side] = record.has_castled_[side];
173  }
174 
175  // マス毎のコピー。
176  for (Square square = 0; square < NUM_SQUARES; square++) {
177  piece_board_[square] = record.piece_board_[square];
178  side_board_[square] = record.side_board_[square];
179  }
180 
181  // 全駒のコピー。
182  blocker_0_ = record.blocker_0_;
183  blocker_45_ = record.blocker_45_;
184  blocker_90_ = record.blocker_90_;
