42  PositionRecord::PositionRecord(const ChessEngine& engine) {
43  // 駒の配置をコピー。ついでにhas_castled_もコピー。
44  for (Side side = 0; side < NUM_SIDES; side++) {
45  has_castled_[side] = engine.has_castled()[side];
46  for (Piece piece_type = 0; piece_type < NUM_PIECE_TYPES; piece_type++) {
47  position_[side][piece_type] = engine.position()[side][piece_type];
48  }
49  }
50 
51  // それ以外をコピー。
52  to_move_ = engine.to_move();
53  castling_rights_ = engine.castling_rights();
54  en_passant_square_ = engine.en_passant_square();
55  ply_100_ = engine.ply_100();

57  pos_hash_ = engine.GetCurrentHash();
58  }

101  bool PositionRecord::operator==(const ChessEngine& engine) const {
102  if (pos_hash_ != engine.GetCurrentHash()) return false;
103  if (to_move_ != engine.to_move()) return false;
104  if (castling_rights_ != engine.castling_rights()) return false;
105  if (en_passant_square_ != engine.en_passant_square()) return false;
106  for (Side side = WHITE; side <= BLACK; side++) {
107  for (Piece piece_type = PAWN; piece_type <= KING; piece_type++) {
108  if (position_[side][piece_type]
109  != engine.position()[side][piece_type]) {
110  return false;
111  }
112  }
113  }
114 
115  return true;
