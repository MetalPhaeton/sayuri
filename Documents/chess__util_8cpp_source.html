143  void Util::InitAttackTable() {
144  // 0度のマップを作成。
145  for (Square square = 0; square < NUM_SQUARES; square++) {
146  Bitboard point = SQUARE[square];
147  point >>= MAGIC_SHIFT_V[square];
148  for (int map = 0; map <= BLOCKER_MAP; map++) {
149  Bitboard attack = 0;
150  // 右側を作る。
151  Bitboard temp = point;
152  while ((temp = GetRightBitboard(temp))) {
153  attack |= temp;
154  if (temp & map) break;
155  }
156  // 左側を作る。
157  temp = point;
158  while ((temp = GetLeftBitboard(temp))) {
159  attack |= temp;
160  if (temp & map) break;
161  }
162  // 利き筋をマスクする。
163  attack &= MAGIC_MASK_V[square];
164  // 利き筋をシフトする。
165  attack <<= MAGIC_SHIFT_V[square];
166  // 利き筋をマップに入れる。
167  attack_table_0_[square][map] = attack;
168  }
169  }

171  for (Square square = 0; square < NUM_SQUARES; square++) {
172  Bitboard point = SQUARE[ROT45[square]];
173  point >>= MAGIC_SHIFT_D[ROT45[square]];

175  Bitboard attack = 0;

565  template<typename CharType>
566  std::vector<std::basic_string<CharType>> Util::Split
567  (const std::basic_string<CharType>& str,
568  const std::set<CharType>& delim,
569  const std::set<CharType>& delim_and_word) {
570  std::vector<std::basic_string<CharType>> ret(0);
571 
572  std::basic_string<CharType> temp;
573  for (auto c : str) {
574  if (delim.find(c) != delim.end()) {
575  if (temp.size() > 0) {
576  ret.push_back(temp);
577  temp.clear();
578  }
579  } else if (delim_and_word.find(c) != delim_and_word.end()) {
580  if (temp.size() > 0) {
581  ret.push_back(temp);
582  temp.clear();
583  temp.push_back(c);
584  ret.push_back(temp);
585  temp.clear();
586  }
587  } else {
588  temp.push_back(c);
589  }
590  }
591 
592  if (temp.size() > 0) {
593  ret.push_back(temp);
594  temp.clear();
595  }
596 
597  return ret;
598  }
