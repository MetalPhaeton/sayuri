187  // FEN文字列を読み込む。
188  void ChessEngine::LoadFen(const Fen& fen) {
189  // キングの数がおかしいならやめる。
190  int num_white_king = Util::CountBits(fen.position()[WHITE][KING]);
191  int num_black_king = Util::CountBits(fen.position()[BLACK][KING]);
192  if ((num_white_king != 1) || (num_black_king != 1)) return;
193 
194  // 空にする。
195  for (Square square = 0; square < NUM_SQUARES; square++) {
196  piece_board_[square] = EMPTY;
197  side_board_[square] = NO_SIDE;
198  }
199  blocker_0_ = 0;
200  blocker_45_ = 0;
201  blocker_90_ = 0;
202  blocker_135_ = 0;
203 
204  // 駒を配置する。
205  for (Side side = WHITE; side <= BLACK; side++) {
206  side_pieces_[side] = 0;
207  for (Piece piece_type = PAWN; piece_type <= KING; piece_type++) {
208  position_[side][piece_type] = fen.position()[side][piece_type];
209  for (Bitboard bb = position_[side][piece_type]; bb; bb &= bb - 1) {
210  Square square = Util::GetSquare(bb);
211  side_board_[square] = side;
212  piece_board_[square] = piece_type;
213  side_pieces_[side] |= Util::SQUARE[square];
214  blocker_0_ |= Util::SQUARE[square];
215  blocker_45_ |= Util::SQUARE[Util::ROT45[square]];
216  blocker_90_ |= Util::SQUARE[Util::ROT90[square]];
217  blocker_135_ |= Util::SQUARE[Util::ROT135[square]];
218  if (piece_type == KING) {
219  king_[side] = square;
220  }
221  }
222  }
223  }
224 
225  // 残りを設定。
226  to_move_ = fen.to_move();
227  castling_rights_ = fen.castling_rights();
228  en_passant_square_ = fen.en_passant_square();
229  ply_100_ = fen.ply_100();
230  ply_ = fen.ply();
231 
232  // 履歴を設定。
233  shared_st_ptr_->ply_100_history_.clear();
234  shared_st_ptr_->ply_100_history_.push_back(ply_100_);
235  shared_st_ptr_->position_history_.clear();
236  shared_st_ptr_->position_history_.push_back(PositionRecord(*this));
237  }

281  // 駒を初期配置にセットする。
282  void ChessEngine::SetStartPosition() {
283  // 駒の配置を作る。
284  // どちらでもない。
285  for (Piece piece_type = 0; piece_type < NUM_PIECE_TYPES; piece_type++) {
286  position_[NO_SIDE][piece_type] = 0;
287  }
288  // 白の駒。
289  position_[WHITE][EMPTY] = 0;
290  position_[WHITE][PAWN] = Util::RANK[RANK_2];
291  position_[WHITE][KNIGHT] = Util::SQUARE[B1] | Util::SQUARE[G1];
292  position_[WHITE][BISHOP] = Util::SQUARE[C1] | Util::SQUARE[F1];
293  position_[WHITE][ROOK] = Util::SQUARE[A1] | Util::SQUARE[H1];
294  position_[WHITE][QUEEN] = Util::SQUARE[D1];
295  position_[WHITE][KING] = Util::SQUARE[E1];
296  // 黒の駒。
297  position_[BLACK][EMPTY] = 0;
298  position_[BLACK][PAWN] = Util::RANK[RANK_7];
299  position_[BLACK][KNIGHT] = Util::SQUARE[B8] | Util::SQUARE[G8];
300  position_[BLACK][BISHOP] = Util::SQUARE[C8] | Util::SQUARE[F8];
301  position_[BLACK][ROOK] = Util::SQUARE[A8] | Util::SQUARE[H8];
302  position_[BLACK][QUEEN] = Util::SQUARE[D8];
303  position_[BLACK][KING] = Util::SQUARE[E8];
304 
305  // 各サイドの駒の配置を作る。
306  side_pieces_[NO_SIDE] = 0;
307  side_pieces_[WHITE] = 0;
308  side_pieces_[BLACK] = 0;
309  for (Piece piece_type = PAWN; piece_type < NUM_PIECE_TYPES; piece_type++) {
310  side_pieces_[WHITE] |= position_[WHITE][piece_type];
311  side_pieces_[BLACK] |= position_[BLACK][piece_type];
312  }
313 
314  // ブロッカーのビットボードを作る。
315  blocker_0_ = side_pieces_[WHITE] | side_pieces_[BLACK];
316  blocker_45_ = 0;
317  blocker_90_ = 0;
318  blocker_135_ = 0;
319  for (Bitboard copy = blocker_0_; copy; copy &= copy - 1) {
320  Square square = Util::GetSquare(copy);
321 
322  blocker_45_ |= Util::SQUARE[Util::ROT45[square]];
323  blocker_90_ |= Util::SQUARE[Util::ROT90[square]];
324  blocker_135_ |= Util::SQUARE[Util::ROT135[square]];
325  }
326 
327  // 駒の種類とサイドの配置を作る。
328  Bitboard point = 1;
329  for (Square square = 0; square < NUM_SQUARES; square++) {
330  // サイドの配置。
331  if (side_pieces_[WHITE] & point) side_board_[square] = WHITE;
332  else if (side_pieces_[BLACK] & point) side_board_[square] = BLACK;
333  else side_board_[square] = NO_SIDE;
334 
335  // 駒の種類の配置。
336  if ((point & position_[WHITE][PAWN])
337  || (point & position_[BLACK][PAWN])) {
338  piece_board_[square] = PAWN;
339  } else if ((point & position_[WHITE][KNIGHT])
340  || (point & position_[BLACK][KNIGHT])) {
341  piece_board_[square] = KNIGHT;
342  } else if ((point & position_[WHITE][BISHOP])
343  || (point & position_[BLACK][BISHOP])) {
344  piece_board_[square] = BISHOP;
345  } else if ((point & position_[WHITE][ROOK])
346  || (point & position_[BLACK][ROOK])) {
347  piece_board_[square] = ROOK;
348  } else if ((point & position_[WHITE][QUEEN])
349  || (point & position_[BLACK][QUEEN])) {
350  piece_board_[square] = QUEEN;
351  } else if ((point & position_[WHITE][KING])
352  || (point & position_[BLACK][KING])) {
353  piece_board_[square] = KING;
354  } else {
355  piece_board_[square] = EMPTY;
356  }
357  point <<= 1;
358  }
359 
360  // キングを入れる。
361  king_[NO_SIDE] = A1; // これは使わない。
362  king_[WHITE] = E1;
363  king_[BLACK] = E8;
364 
365  // 手番を初期化。
366  to_move_ = WHITE;
367 
368  // キャスリングの権利を初期化。
369  castling_rights_ = ALL_CASTLING;
370 
371  // アンパッサンを初期化。
372  en_passant_square_ = 0;
373 
374  // 50手ルールを初期化。
375  ply_100_ = 0;
376 
377  // 手数を初期化。
378  ply_ = 1;
379 
380  // キャスリングしたかどうかの初期化。
381  for (Side side = 0; side < NUM_SIDES; side++) {
382  has_castled_[side] = false;
383  }
384 
385  if (shared_st_ptr_) {
386  // 50手ルールの履歴を初期化。
387  shared_st_ptr_->ply_100_history_.clear();
388  shared_st_ptr_->ply_100_history_.push_back(0);
389 
390  // 駒の配置の履歴を初期化。
391  shared_st_ptr_->position_history_.clear();
392  shared_st_ptr_->position_history_.push_back(PositionRecord(*this));
393  }
394  }

482  // 手を指す。
483  void ChessEngine::PlayMove(Move move) {
484  // 合法手かどうか調べる。
485  // 手を展開する。
486  shared_st_ptr_->history_max_ = 1; // makerが0の除算をしないように。
487  MoveMaker maker(*this);
488  maker.GenMoves<GenMoveType::ALL>(0, 0, 0, 0);
489  // 合法手かどうか調べる。
490  bool is_legal = false;
491  Side side = to_move_;
492  Side enemy_side = side ^ 0x3;
493  for (Move temp_move = maker.PickMove(); temp_move;
494  temp_move = maker.PickMove()) {
495  MakeMove(temp_move);
496  // temp_moveが合法手かどうか調べる。
497  if (IsAttacked(king_[side], enemy_side)) {
498  UnmakeMove(temp_move);
499  continue;
500  }
501  // temp_moveと同じ手かどうか調べる。
502  if (EqualMove(move, temp_move)) {
503  UnmakeMove(temp_move);
504  move = temp_move;
505  is_legal = true;
506  break;
507  }
508  UnmakeMove(temp_move);
509  }
510 
511  if (is_legal) {
512  ply_++;
513  if ((piece_board_[move_from(move)] == PAWN)
514  || (piece_board_[move_to(move)] != EMPTY)) {
515  ply_100_ = 0;
516  } else {
517  ply_100_++;
518  }
519  shared_st_ptr_->move_history_.push_back(move);
520  shared_st_ptr_->ply_100_history_.push_back(ply_100_);
521  MakeMove(move);
522  shared_st_ptr_->position_history_.push_back(PositionRecord(*this));
523  }
524  }

541  // キャスリング出来るかどうかを判定する。
542  template<Castling Flag>
543  bool ChessEngine::CanCastling() const {
544  if (!(castling_rights_ & Flag)) return false;
545 
546  if (Flag == WHITE_SHORT_CASTLING) {
547  if (king_[WHITE] != E1) return false;
548  if (!(position_[WHITE][ROOK] & Util::SQUARE[H1])) return false;
549  if (IsAttacked(E1, BLACK)) return false;
550  if (IsAttacked(F1, BLACK)) return false;
551  if (IsAttacked(G1, BLACK)) return false;
552  if (piece_board_[F1]) return false;
553  if (piece_board_[G1]) return false;
554  } else if (Flag == WHITE_LONG_CASTLING) {
555  if (king_[WHITE] != E1) return false;
556  if (!(position_[WHITE][ROOK] & Util::SQUARE[A1])) return false;
557  if (IsAttacked(E1, BLACK)) return false;
558  if (IsAttacked(D1, BLACK)) return false;
559  if (IsAttacked(C1, BLACK)) return false;
560  if (piece_board_[D1]) return false;
561  if (piece_board_[C1]) return false;
562  if (piece_board_[B1]) return false;
563  } else if (Flag == BLACK_SHORT_CASTLING) {
564  if (king_[BLACK] != E8) return false;
565  if (!(position_[BLACK][ROOK] & Util::SQUARE[H8])) return false;
566  if (IsAttacked(E8, WHITE)) return false;
567  if (IsAttacked(F8, WHITE)) return false;
568  if (IsAttacked(G8, WHITE)) return false;
569  if (piece_board_[F8]) return false;
570  if (piece_board_[G8]) return false;

580  } else {

592  // 次の手を指す。
593  void ChessEngine::MakeMove(Move& move) {
594  // 動かす側のサイドを得る。
595  Side side = to_move_;
596 
597  // 手番を反転させる。
598  to_move_ = to_move_ ^ 0x3;
599 
600  // 動かす前のキャスリングの権利とアンパッサンを記録する。
601  move_castling_rights(move, castling_rights_);
602  move_en_passant_square(move, en_passant_square_);
603 
604  // アンパッサンを解除。
605  en_passant_square_ = 0;
606 
607  // 手の要素を得る。
608  Square from = move_from(move);
609  Square to = move_to(move);
610  Piece promotion = move_promotion(move);
611  MoveType move_type = move_move_type(move);
612 
613  // NULL_MOVEならNull Moveする。
614  if (move_type == NULL_MOVE) {
615  return;
616  }
617 
618  // キャスリングの権利を更新。
619  Piece piece = piece_board_[from];
620  if (side == WHITE) {
621  if (piece == KING) {
622  castling_rights_ &= ~WHITE_CASTLING;
623  } else if (piece == ROOK) {
624  if (from == H1) {
625  castling_rights_ &= ~WHITE_SHORT_CASTLING;
626  } else if (from == A1) {
627  castling_rights_ &= ~WHITE_LONG_CASTLING;
628  }
629  }
630  } else {
631  if (piece == KING) {
632  castling_rights_ &= ~BLACK_CASTLING;
633  } else if (piece == ROOK) {
634  if (from == H8) {
635  castling_rights_ &= ~BLACK_SHORT_CASTLING;
636  } else if (from == A8) {
637  castling_rights_ &= ~BLACK_LONG_CASTLING;
638  }
639  }
640  }
641 
642  // 手の種類によって分岐する。
643  if (move_type == CASTLING) { // キャスリングの場合。
644  // キングを動かす。
645  ReplacePiece(from, to);
646  // ルークを動かす。
647  if (to == G1) {
648  ReplacePiece(H1, F1);
649  } else if (to == C1) {
650  ReplacePiece(A1, D1);
651  } else if (to == G8) {
652  ReplacePiece(H8, F8);
653  } else if (to == C8) {
654  ReplacePiece(A8, D8);
655  }
656  has_castled_[side] = true;
657  } else if (move_type == EN_PASSANT) { // アンパッサンの場合。
658  // 取った駒をボーンにする。
659  move_captured_piece(move, PAWN);
660  // 動かす。
661  ReplacePiece(from, to);
662  // アンパッサンのターゲットを消す。
663  Square en_passant_target =
664  side == WHITE ? to - 8 : to + 8;
665  PutPiece(en_passant_target, EMPTY);
666  } else { // それ以外の場合。
667  // 取る駒を登録する。
668  move_captured_piece(move, piece_board_[to]);
669  // 駒を動かす。
670  ReplacePiece(from, to);
671  // 駒を昇格させるなら、駒を昇格させる。
672  if (promotion) {
673  PutPiece(to, promotion, side);
674  }
675  // ポーンの2歩の動きの場合はアンパッサンできるようにする。
676  if (piece_board_[to] == PAWN) {
677  if (((side == WHITE) && ((from + 16) == to))
678  || ((side == BLACK) && ((from - 16) == to))) {
679  en_passant_square_ = side == WHITE ? to - 8 : to + 8;
680  }
681  }
682  }
683  }

691  to_move_ ^= 0x3;
692 
693  // 動かす前のキャスリングの権利とアンパッサンを復元する。

695  en_passant_square_ = move_en_passant_square(move);

702

704  if (move_type == NULL_MOVE) {

713  // ルークを戻す。
714  if (to == G1) {

716  } else if (to == C1) {

743  Bitboard attack = Util::GetPawnAttack(square, side ^ 0x3);
744  if (attack & position_[side][PAWN]) return true;

757  if (attack & (position_[side][ROOK] | position_[side][QUEEN]))
758  return true;

761  attack = Util::GetKingMove(square);
762  if (attack & position_[side][KING]) return true;

765  return false;

830  // 次の局面のハッシュを得る。
831  Hash ChessEngine::GetNextHash(Hash current_hash, Move move) const {
832  // 駒の情報を得る。
833  Square from = move_from(move);
834  Square to = move_to(move);
835  Piece promotion = move_promotion(move);
836 
837  // 駒の位置の種類とサイドを得る。
838  Piece piece_type = piece_board_[from];
839  Side piece_side = side_board_[from];
840 
841  // 取る駒の種類とサイドを得る。
842  Piece target_type = piece_board_[to];
843  Side target_side = side_board_[to];
844  Square target_square = to;
845  if ((piece_type == PAWN) && en_passant_square_
846  && (to == en_passant_square_)) {
847  // アンパッサンの時。
848  if (piece_side == WHITE) {
849  target_square = to - 8;
850  } else {
851  target_square = to + 8;
852  }
853  target_type = piece_board_[target_square];
854  }
855 
856  // 移動する駒のハッシュを削除する。
857  current_hash ^=
858  shared_st_ptr_->piece_hash_value_table_[piece_side][piece_type][from];
859 
860  // 取る駒のハッシュを削除する。
861  current_hash ^= shared_st_ptr_->piece_hash_value_table_
862  [target_side][target_type][target_square];
863 
864  // 移動する駒の移動先のハッシュを追加する。
865  if (promotion) {
866  current_hash ^=
867  shared_st_ptr_->piece_hash_value_table_[piece_side][promotion][to];
868  } else {
869  current_hash ^=
870  shared_st_ptr_->piece_hash_value_table_[piece_side][piece_type][to];
871  }
872 
873  // 現在の手番のハッシュを削除。
874  current_hash ^= shared_st_ptr_->to_move_hash_value_table_[to_move_];
875 
876  // 次の手番のハッシュを追加。
877  current_hash ^= shared_st_ptr_->to_move_hash_value_table_[to_move_ ^ 0x3];
878 
879  // キャスリングのハッシュをセット。
880  Castling loss_rights = 0;
881  if (piece_side == WHITE) {
882  if (piece_type == KING) {
883  loss_rights |= WHITE_CASTLING;
884  } else if (piece_type == ROOK) {
885  if (from == H1) {
886  loss_rights |= WHITE_SHORT_CASTLING;
887  } else if (from == A1) {
888  loss_rights |= WHITE_LONG_CASTLING;
889  }
890  }
891  } else {
892  if (piece_type == KING) {
893  loss_rights |= BLACK_CASTLING;
894  } else if (piece_type == ROOK) {
895  if (from == H8) {
896  loss_rights |= BLACK_SHORT_CASTLING;
897  } else if (from == A8) {
898  loss_rights |= BLACK_LONG_CASTLING;
899  }
900  }
901  }
902  Castling castling_diff =
903  (castling_rights_ & ~(loss_rights)) ^ castling_rights_;
904  Castling bit = 1;
905  for (int i = 0; i < 4; i++) {
906  if (castling_diff & bit) {
907  current_hash ^= shared_st_ptr_->castling_hash_value_table_[i];
908  }
909  bit <<= 1;
910  }
911 
912  // とりあえずアンパッサンのハッシュを削除。
913  current_hash ^=
914  shared_st_ptr_->en_passant_hash_value_table_[en_passant_square_];
915 
916  // ポーンの2歩の動きの場合はアンパッサンハッシュを追加。
917  if (piece_type == PAWN) {
918  int move_diff = to - from;
919  if (move_diff == 16) {
920  current_hash ^= shared_st_ptr_->en_passant_hash_value_table_[to - 8];
921  } else if (move_diff == -16) {
922  current_hash ^= shared_st_ptr_->en_passant_hash_value_table_[to + 8];
923  }
924  }
925 
926  // 次の局面のハッシュを返す。
927  return current_hash;
928  }

933  // 駒を置く。
934  void ChessEngine::PutPiece(Square square, Piece piece_type, Side side) {
935  // 置く位置の現在の駒の種類を入手する。
936  Piece placed_piece = piece_board_[square];
937 
938  // 置く位置の現在の駒のサイドを得る。
939  Side placed_side = side_board_[square];
940 
941  // 置く位置のメンバを消す。
942  if (placed_piece) {
943  position_[placed_side][placed_piece] &= ~Util::SQUARE[square];
944  side_pieces_[placed_side] &= ~Util::SQUARE[square];
945  }
946 
947  // 置く駒がEMPTYか置くサイドがNO_SIDEなら
948  // その位置のメンバを消して返る。
949  if ((!piece_type) || (!side)) {
950  piece_board_[square] = EMPTY;
951  side_board_[square] = NO_SIDE;
952  if (placed_piece) {
953  blocker_0_ &= ~Util::SQUARE[square];
954  blocker_45_ &= ~Util::SQUARE[Util::ROT45[square]];
955  blocker_90_ &= ~Util::SQUARE[Util::ROT90[square]];
956  blocker_135_ &= ~Util::SQUARE[Util::ROT135[square]];
957  }
958  return;
959  }
960 
961  // 置く位置の駒の種類を書き変える。
962  piece_board_[square] = piece_type;
963  // 置く位置のサイドを書き変える。
964  side_board_[square] = side;
965 
966  // 置く位置のビットボードをセットする。
967  position_[side][piece_type] |= Util::SQUARE[square];
968  side_pieces_[side] |= Util::SQUARE[square];
969  blocker_0_ |= Util::SQUARE[square];
970  blocker_45_ |= Util::SQUARE[Util::ROT45[square]];
971  blocker_90_ |= Util::SQUARE[Util::ROT90[square]];
972  blocker_135_ |= Util::SQUARE[Util::ROT135[square]];
973 
974  // キングの位置を更新する。
975  if (piece_type == KING) {
976  king_[side] = square;
977  }
978  }
