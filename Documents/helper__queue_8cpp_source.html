77  // スレッドが仕事を得る。
78  Job* HelperQueue::GetJob() {
79  std::unique_lock<std::mutex> lock(mutex_); // ロック。
80 
81  // ヘルパーが必要なければnullptrを返す。
82  if (no_more_help_) {
83  client_cond_.notify_all();
84  return nullptr;
85  }
86 
87  // 待つ。
88  // ルートノードのクライアントが待っていれば待たない。
89  if (!is_root_client_waiting_) {
90  while (!job_ptr_) {
91  num_helpers_++;
92  helper_cond_.wait(lock);
93  num_helpers_--;
94  // 待っている間にno_more_help_が立っているかもしれないので、
95  // もう一度確認。
96  // ヘルパーが必要なければnullptrを返す。
97  if (no_more_help_) {
98  client_cond_.notify_all();
99  return nullptr;
100  }
101  }
102  }
103  is_root_client_waiting_ = false;
104 
105 
106  // 仕事の準備。
107  job_ptr_->CountHelper();
108  Job* temp = job_ptr_;
109  job_ptr_ = nullptr;
110 
111  // 準備完了を通知。
112  client_cond_.notify_all();
113 
114  return temp;
115  }
