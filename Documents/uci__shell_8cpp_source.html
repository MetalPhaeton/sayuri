227  sout << " nodes " << num_nodes;

233  sout << " " << TransMoveToString(pv_line[i]);

235

237  for (auto& func : output_listeners_) {

412  // 「setoption」コマンドのコールバック関数。
413  void UCIShell::CommandSetOption(UCICommand::CommandArgs& args) {
414  // nameとvalueがあるかどうか。
415  // なければ設定できない。
416  if ((args.find("name") == args.end())
417  || (args.find("value") == args.end())) {
418  return;
419  }
420 
421  // nameの文字列をくっつける。
422  std::string name_str = "";
423  for (unsigned int i = 1; i < args["name"].size(); i++) {
424  name_str += args["name"][i] + " ";
425  }
426  name_str.pop_back();
427 
428  // nameの文字を全部小文字にする。
429  for (auto& c : name_str) c = std::tolower(c);
430 
431  // nameごとの処理。
432  if (name_str == "hash") {
433  // トランスポジションテーブルのサイズ変更。
434  try {
435  table_size_ = std::stoull(args["value"][1]) * 1024ULL * 1024ULL;
436 
437  table_size_ = table_size_ >= UCI_MIN_TABLE_SIZE
438  ? table_size_ : UCI_MIN_TABLE_SIZE;
439 
440  table_size_ = table_size_ <= UCI_MAX_TABLE_SIZE
441  ? table_size_ : UCI_MAX_TABLE_SIZE;
442 
443  table_ptr_.reset(new TranspositionTable(table_size_));
444  } catch (...) {
445  // 無視。
446  }
447  } else if (name_str == "clear hash") {
448  // トランスポジションテーブルの初期化。
449  table_ptr_.reset(new TranspositionTable(table_size_));
450  } else if (name_str == "ponder") {
451  // Ponderの有効化、無効化。
452  if (args["value"][1] == "true") enable_pondering_ = true;
453  else if (args["value"][1] == "false") enable_pondering_ = false;
454  } else if (name_str == "threads") {
455  // スレッドの数の変更。
456  try {
457  num_threads_ = std::stoi(args["value"][1]);
458 
459  num_threads_ = num_threads_ >= 1 ? num_threads_ : 1;
460  num_threads_ = num_threads_ <= UCI_MAX_THREADS
461  ? num_threads_ : UCI_MAX_THREADS;
462  } catch (...) {
463  // 無視。
464  }
465  } else if (name_str == "uci_analysemode") {
466  // アナライズモードの有効化、無効化。
467  if (args["value"][1] == "true") analyse_mode_ = true;
468  else if (args["value"][1] == "false") analyse_mode_ = false;
469  }
470  }
