45  Fen::Fen(const std::string fen_str) :
46  en_passant_square_(0),
47  ply_100_(0),
48  ply_(1) {
49  // fen_strを分解。
50  std::vector<std::string> fen_tokens = Util::Split(fen_str, " ", "");
51 
52  try {
53  // FENを解析。
54  EvalPosition(fen_tokens[0]);
55  EvalToMove(fen_tokens[1]);
56  EvalCastlingRights(fen_tokens[2]);
57  if (fen_tokens.size() >= 4) EvalEnPassant(fen_tokens[3]);
58  if (fen_tokens.size() >= 5) EvalPly100(fen_tokens[4]);
59  if (fen_tokens.size() >= 6) EvalPly(fen_tokens[5]);
60  } catch (...) {
61  SetStartPosition();
62  }
63  }

137  // 駒の配置文字列を評価する。
138  void Fen::EvalPosition(const std::string& position_str) {
139  // 駒の配置を初期化。
140  for (Side side = 0; side < NUM_SIDES; side++) {
141  for (Piece piece_type = 0; piece_type < NUM_PIECE_TYPES; piece_type++) {
142  position_[side][piece_type] = 0;
143  }
144  }
145 
146  // ランク毎に区切って、逆転。
147  // fenは8ランクから始まるので、1ランクからに逆転する必要がある。
148  std::vector<std::string> position_vec = Util::Split(position_str, "/", "");
149  std::reverse(position_vec.begin(), position_vec.end());
150 
151  // 値を格納していく。
152  int square = 0; // マスの位置。
153  for (auto& a : position_vec) {
154  for (auto& b : a) {
155 
156  // 文字を解析。
157  switch (b) {
158  case '1':
159  case '2':
160  case '3':
161  case '4':
162  case '5':
163  case '6':
164  case '7':
165  case '8':
166  square += (b - '0') - 1;
167  break;
168  case 'P':
169  position_[WHITE][PAWN] |= Util::SQUARE[square];
170  break;
171  case 'N':
172  position_[WHITE][KNIGHT] |= Util::SQUARE[square];
173  break;
174  case 'B':
175  position_[WHITE][BISHOP] |= Util::SQUARE[square];
176  break;
177  case 'R':
178  position_[WHITE][ROOK] |= Util::SQUARE[square];
179  break;
180  case 'Q':
181  position_[WHITE][QUEEN] |= Util::SQUARE[square];
182  break;
183  case 'K':
184  position_[WHITE][KING] |= Util::SQUARE[square];
185  break;
186  case 'p':
187  position_[BLACK][PAWN] |= Util::SQUARE[square];
188  break;
189  case 'n':
190  position_[BLACK][KNIGHT] |= Util::SQUARE[square];
191  break;
192  case 'b':
193  position_[BLACK][BISHOP] |= Util::SQUARE[square];
194  break;
195  case 'r':
196  position_[BLACK][ROOK] |= Util::SQUARE[square];
197  break;
198  case 'q':
199  position_[BLACK][QUEEN] |= Util::SQUARE[square];
200  break;
201  case 'k':
202  position_[BLACK][KING] |= Util::SQUARE[square];
203  break;
204  default:
205  throw SayuriError("FENを解析できません。");
206  break;
207  }
208  square++;
209  if (square > 64) {
210  throw SayuriError("FENを解析できません。");
211  }
212  }
213  }
214  }

231  // キャスリングの権利文字列を評価する。
232  void Fen::EvalCastlingRights(const std::string& castling_rights_str) {
233  castling_rights_ = 0;
234 
235  if (castling_rights_str[0] == '-') return;
236 
237  for (auto& a : castling_rights_str) {
238  switch (a) {
239  case 'K':
240  castling_rights_ |= WHITE_SHORT_CASTLING;
241  break;
242  case 'Q':
243  castling_rights_ |= WHITE_LONG_CASTLING;
244  break;
245  case 'k':
246  castling_rights_ |= BLACK_SHORT_CASTLING;
247  break;
248  case 'q':
249  castling_rights_ |= BLACK_LONG_CASTLING;
250  break;
251  default:
252  throw SayuriError("FENを解析できません。");
253  break;
254  }
255  }

258  // アンパッサン文字列を評価する。
259  void Fen::EvalEnPassant(const std::string& en_passant_square_str) {
260  // アンパッサンがない。
261  if (en_passant_square_str[0] == '-') {
262  en_passant_square_ = 0;
263  return;
264  }
265 
266  // ファイルから評価。
267  int index = en_passant_square_str[0] - 'a';
268  if ((index < 0) || (index > 7)) {
269  throw SayuriError("FENを解析できません。");
270  }
271  Bitboard fyle = Util::FYLE[index];
272 
273  // ランクを評価。
274  if (en_passant_square_str[1] == '3') {
275  index = RANK_3;
276  } else if (en_passant_square_str[1] == '6') {
277  index = RANK_6;
278  } else {
279  throw SayuriError("FENを解析できません。");
280  }
281  Bitboard rank = Util::RANK[index];
282 
283  en_passant_square_ = Util::GetSquare(fyle & rank);
284  }
