102  // 現在の局面の評価値を計算する。
103  int Evaluator::Evaluate(int material) {
104  // 価値の変数の初期化。
105  for (Piece piece_type = PAWN; piece_type <= KING; piece_type++) {
106  opening_position_value_[piece_type] = 0.0;
107  ending_position_value_[piece_type] = 0.0;
108  mobility_value_[piece_type] = 0.0;
109  center_control_value_[piece_type] = 0.0;
110  sweet_center_control_value_[piece_type] = 0.0;
111  development_value_[piece_type] = 0.0;
112  attack_value_[piece_type] = 0.0;
113  defense_value_[piece_type] = 0.0;
114  pin_value_[piece_type] = 0.0;
115  attack_around_king_value_[piece_type] = 0.0;
116  }
117  pass_pawn_value_ = 0.0;
118  protected_pass_pawn_value_ = 0.0;
119  double_pawn_value_ = 0.0;
120  iso_pawn_value_ = 0.0;
121  pawn_shield_value_ = 0.0;
122  bishop_pair_value_ = 0.0;
123  bad_bishop_value_ = 0.0;
124  pin_knight_value_ = 0.0;
125  rook_pair_value_ = 0.0;
126  rook_semiopen_fyle_value_ = 0.0;
127  rook_open_fyle_value_ = 0.0;
128  early_queen_launched_value_ = 0.0;
129  weak_square_value_ = 0.0;
130  castling_value_ = 0.0;
131  abandoned_castling_value_ = 0.0;
132 
133  // サイド。
134  Side side = engine_ptr_->to_move();
135  Side enemy_side = side ^ 0x3;
136 
137  // 十分な駒がない場合は引き分け。
138  if (!HasEnoughPieces(side) && !HasEnoughPieces(enemy_side)) {
139  return SCORE_DRAW;
140  }
141 
142  // 全体計算。
143  // ビショップペア。
144  if (Util::CountBits(engine_ptr_->position()[side][BISHOP]) >= 2) {
145  bishop_pair_value_ += 1.0;
146  }
147  if (Util::CountBits(engine_ptr_->position()[enemy_side][BISHOP]) >= 2) {
148  bishop_pair_value_ -= 1.0;
149  }
150  // ルークペア。
151  if (Util::CountBits(engine_ptr_->position()[side][ROOK]) >= 2) {
152  rook_pair_value_ += 1.0;
153  }
154  if (Util::CountBits(engine_ptr_->position()[enemy_side][ROOK]) >= 2) {
155  rook_pair_value_ -= 1.0;
156  }
157 
158  // 各駒毎に価値を計算する。
159  Bitboard all_pieces = engine_ptr_->blocker_0();
160  for (Bitboard pieces = all_pieces; pieces; pieces &= pieces - 1) {
161  Square piece_square = Util::GetSquare(pieces);
162  Side piece_side = engine_ptr_->side_board()[piece_square];
163  switch (engine_ptr_->piece_board()[piece_square]) {
164  case PAWN:
165  CalValue<PAWN>(piece_square, piece_side);
166  break;
167  case KNIGHT:
168  CalValue<KNIGHT>(piece_square, piece_side);
169  break;
170  case BISHOP:
171  CalValue<BISHOP>(piece_square, piece_side);
172  break;
173  case ROOK:
174  CalValue<ROOK>(piece_square, piece_side);
175  break;
176  case QUEEN:
177  CalValue<QUEEN>(piece_square, piece_side);
178  break;
179  case KING:
180  CalValue<KING>(piece_square, piece_side);
181  break;
182  default:
183  throw SayuriError("駒の種類が不正です。");
184  break;
185  }
186  }
187 
188  // ウェイトを付けて評価値を得る。
189  constexpr double NUM_KINGS = 2.0;
190  double num_pieces = Util::CountBits(all_pieces) - NUM_KINGS;
191  const EvalParams& params = engine_ptr_->eval_params();
192 
193  // マテリアル。
194  double score = material;
195 
196  // 配列型のウェイトの評価。
197  for (Piece piece_type = PAWN; piece_type <= KING; piece_type++) {
198  score +=
199  // オープニング時の駒の配置。
200  (params.weight_opening_position()[piece_type](num_pieces)
201  * opening_position_value_[piece_type])
202  // エンディング時の駒の配置。
203  + (params.weight_ending_position()[piece_type](num_pieces)
204  * ending_position_value_[piece_type])
205  // 機動力。
206  + (params.weight_mobility()[piece_type](num_pieces)
207  * mobility_value_[piece_type])
208  // センターコントロール。
209  + (params.weight_center_control()[piece_type](num_pieces)
210  * center_control_value_[piece_type])
211  // スウィートセンターのコントロール。
212  + (params.weight_sweet_center_control()[piece_type](num_pieces)
213  * sweet_center_control_value_[piece_type])
214  // 駒の展開。
215  + (params.weight_development()[piece_type](num_pieces)
216  * development_value_[piece_type])
217  // 攻撃。
218  + (params.weight_attack()[piece_type](num_pieces)
219  * attack_value_[piece_type])
220  // 防御。
221  + (params.weight_defense()[piece_type](num_pieces)
222  * defense_value_[piece_type])
223  // ピン。
224  + (params.weight_pin()[piece_type](num_pieces)
225  * pin_value_[piece_type])
226  // 相手キング周辺への攻撃。
227  + (params.weight_attack_around_king()[piece_type](num_pieces)
228  * attack_around_king_value_[piece_type]);
229  }
230 
231  // その他のウェイトの評価。
232  score +=
233  // パスポーン。
234  + (params.weight_pass_pawn()(num_pieces) * pass_pawn_value_)
235  // 守られたパスポーン。
236  + (params.weight_protected_pass_pawn()(num_pieces)
237  * protected_pass_pawn_value_)
238  // ダブルポーン。
239  + (params.weight_double_pawn()(num_pieces) * double_pawn_value_)
240  // 孤立ポーン。
241  + (params.weight_iso_pawn()(num_pieces) * iso_pawn_value_)
242  // ポーンの盾。
243  + (params.weight_pawn_shield()(num_pieces) * pawn_shield_value_)
244  // ビショップペア。
245  + (params.weight_bishop_pair()(num_pieces) * bishop_pair_value_)
246  // バッドビショップ。
247  + (params.weight_bad_bishop()(num_pieces) * bad_bishop_value_)
248  // ルークペア。
249  + (params.weight_rook_pair()(num_pieces) * rook_pair_value_)
250  // セミオープンファイルのルーク。
251  + (params.weight_rook_semiopen_fyle()(num_pieces)
252  * rook_semiopen_fyle_value_)
253  // オープンファイルのルーク。
254  + (params.weight_rook_open_fyle()(num_pieces) * rook_open_fyle_value_)
255  // 早すぎるクイーンの始動。
256  + (params.weight_early_queen_launched()(num_pieces)
257  * early_queen_launched_value_)
258  // キング周りの弱いマス。
259  + (params.weight_weak_square()(num_pieces) * weak_square_value_)
260  // キャスリング。
261  + (params.weight_castling()(num_pieces) * castling_value_)
262  // キャスリングの放棄。
263  + (params.weight_abandoned_castling()(num_pieces)
264  * abandoned_castling_value_);
265 
266  return static_cast<int>(score);
267  }

397  bool Evaluator::HasEnoughPieces(Side side) const {
398  // ポーンがあれば大丈夫。
399  if (engine_ptr_->position()[side][PAWN]) return true;
400 
401  // ルークがあれば大丈夫。
402  if (engine_ptr_->position()[side][ROOK]) return true;
403 
404  // クイーンがあれば大丈夫。
405  if (engine_ptr_->position()[side][QUEEN]) return true;
406 
407  // ビショップが2つあれば大丈夫。
408  if (Util::CountBits(engine_ptr_->position()[side][BISHOP]) >= 2)
409  return true;
410 
411  // ナイトが2つあれば大丈夫。
412  if (Util::CountBits(engine_ptr_->position()[side][KNIGHT]) >= 2)
413  return true;
414 
415  // ナイトとビショップの合計が2つあれば大丈夫。
416  if (Util::CountBits(engine_ptr_->position()[side][KNIGHT]
417  | engine_ptr_->position()[side][BISHOP]) >= 2)
418  return true;
419 
420  // それ以外はダメ。
421  return false;
422  }

424  // ================== //
425  // 価値を計算する関数 //
426  // ================== //
427  // 各駒の価値を計算する。
428  template<Piece Type>
429  void Evaluator::CalValue(Square piece_square, Side piece_side) {
430  // サイド。
431  Side enemy_piece_side = piece_side ^ 0x3;
432 
433  // 値と符号。自分の駒ならプラス。敵の駒ならマイナス。
434  double value;
435  double sign = piece_side == engine_ptr_->to_move() ? 1.0 : -1.0;
436 
437  // 評価関数用パラメータを得る。
438  const EvalParams& params = engine_ptr_->eval_params();
439 
440  // 利き筋を作る。
441  Bitboard attacks = 0;
442  Bitboard pawn_moves = 0;
443  Bitboard en_passant = 0;
444  Bitboard castling_moves = 0;
445  switch (Type) {
446  case PAWN:
447  // 通常の動き。
448  pawn_moves = Util::GetPawnMove(piece_square, piece_side)
449  & ~(engine_ptr_->blocker_0());
450  // 2歩の動き。
451  if (pawn_moves) {
452  if (((piece_side == WHITE)
453  && (Util::GetRank(piece_square) == RANK_2))
454  || ((piece_side == BLACK)
455  && (Util::GetRank(piece_square) == RANK_7))) {
456  // ポーンの2歩の動き。
457  pawn_moves |= Util::GetPawn2StepMove(piece_square, piece_side)
458  & ~(engine_ptr_->blocker_0());
459  }
460  }
461  // 攻撃。
462  attacks = Util::GetPawnAttack(piece_square, piece_side);
463 
464  // アンパッサン。
465  if (engine_ptr_->en_passant_square()
466  && (piece_side == engine_ptr_->to_move())) {
467  en_passant =
468  Util::SQUARE[engine_ptr_->en_passant_square()] & attacks;
469  }
470  break;
471  case KNIGHT:
472  attacks = Util::GetKnightMove(piece_square);
473  break;
474  case BISHOP:
475  attacks = engine_ptr_->GetBishopAttack(piece_square);
476  break;
477  case ROOK:
478  attacks = engine_ptr_->GetRookAttack(piece_square);
479  break;
480  case QUEEN:
481  attacks = engine_ptr_->GetQueenAttack(piece_square);
482  break;
483  case KING:
484  attacks = Util::GetKingMove(piece_square);
485  castling_moves = 0;
486  // キャスリングの動きを追加。
487  if (piece_side == WHITE) {
488  if (engine_ptr_->CanCastling<WHITE_SHORT_CASTLING>()) {
489  castling_moves |= Util::SQUARE[G1];
490  }
491  if (engine_ptr_->CanCastling<WHITE_LONG_CASTLING>()) {
492  castling_moves |= Util::SQUARE[C1];
493  }
494  } else {
495  if (engine_ptr_->CanCastling<BLACK_SHORT_CASTLING>()) {
496  castling_moves |= Util::SQUARE[G8];
497  }
498  if (engine_ptr_->CanCastling<BLACK_LONG_CASTLING>()) {
499  castling_moves |= Util::SQUARE[C8];
500  }
501  }
502  break;
503  default:
504  throw SayuriError("駒の種類が不正です。");
505  break;
506  }
507 
508  // --- 全駒共通 --- //
509  // オープニング時の駒の配置を計算。
510  if (piece_side == WHITE) {
511  opening_position_value_[Type] += sign
512  * params.opening_position_value_table()[Type][piece_square];
513  } else {
514  opening_position_value_[Type] += sign
515  * params.opening_position_value_table()[Type][Util::FLIP[piece_square]];
516  }
517 
518  // エンディング時の駒の配置を計算。
519  if (piece_side == WHITE) {
520  ending_position_value_[Type] += sign
521  * params.ending_position_value_table()[Type][piece_square];
522  } else {
523  ending_position_value_[Type] += sign
524  * params.ending_position_value_table()[Type][Util::FLIP[piece_square]];
525  }
526 
527  // 機動力を計算。
528  if (Type == PAWN) {
529  mobility_value_[Type] += sign * Util::CountBits((attacks
530  & engine_ptr_->side_pieces()[enemy_piece_side])
531  | pawn_moves | en_passant);
532  } else {
533  mobility_value_[Type] += sign * Util::CountBits(attacks
534  & ~(engine_ptr_->side_pieces()[piece_side]));
535  }
536 
537  // センターコントロールを計算。
538  center_control_value_[Type] +=
539  sign * Util::CountBits(attacks & center_mask_);
540 
541  // スウィートセンターのコントロールを計算。
542  sweet_center_control_value_[Type] += sign
543  * Util::CountBits(attacks & sweet_center_mask_);
544 
545  // 駒の展開を計算。
546  if (!(Util::SQUARE[piece_square] & start_position_[piece_side][Type])) {
547  development_value_[Type] += sign * 1.0;
548  }
549 
550  // 敵への攻撃を計算。
551  Bitboard attacked =
552  attacks & (engine_ptr_->side_pieces()[enemy_piece_side]);
553  value = 0.0;

566  Bitboard defensed = attacks & (engine_ptr_->side_pieces()[piece_side]);
567  value = 0.0;

570  for (; defensed; defensed &= defensed - 1) {

572  table_2[Type][engine_ptr_->piece_board()[Util::GetSquare(defensed)]];

577  if ((Type == BISHOP) || (Type == ROOK) || (Type == QUEEN)) {
578  value = 0.0;

580  // ピンの対象候補。
581  Bitboard pin_target =
582  attacks & engine_ptr_->side_board()[enemy_piece_side];

590  } else if (Type == ROOK) {
591  pin_back =

593  & Util::GetRookMove(piece_square)) & ~pin_target;
594  } else {
595  pin_back =

597  & Util::GetQueenMove(piece_square)) & ~pin_target;

603  Square pin_back_sq = Util::GetSquare(bb);
604 
605  // 裏駒と自分の間のビットボード。
606  Bitboard between = Util::GetBetween(piece_square, pin_back_sq);

612  [engine_ptr_->piece_board()[Util::GetSquare(between & pin_target)]]
613  [engine_ptr_->piece_board()[pin_back_sq]];

622  & Util::GetKingMove(engine_ptr_->king()[enemy_piece_side]));

628  if (!(engine_ptr_->position()[enemy_piece_side][PAWN]

643  }

646  if (!(engine_ptr_->position()[piece_side][PAWN]

696  & start_position_[piece_side][KNIGHT]);

706  // 弱いマス。
707  value = 0.0;

714  value += Util::CountBits(engine_ptr_->position()

731  abandoned_castling_value_ += sign * 1.0;
