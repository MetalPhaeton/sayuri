137  // 次の候補手を取り出す。
138  Move MoveMaker::PickMove() {
139  std::unique_lock<std::mutex> lock(mutex_);
140 
141  MoveSlot slot;
142 
143  // 手がなければ何もしない。
144  if (!last_) {
145  return 0;
146  }
147 
148  // とりあえず最後の手をポップ。
149  last_--;
150  slot = move_stack_[last_];
151 
152  // 一番高い手を探し、スワップ。
153  for (int i = last_ - 1; i >= 0; i--) {
154  if (move_stack_[i].score_ > slot.score_) {
155  std::swap(slot, move_stack_[i]);
156  }
157  }
158 
159  return slot.move_;
160  }

162  // スタックに候補手を生成する。 (内部用)
163  template<GenMoveType Type>
164  void MoveMaker::GenMovesCore(Move prev_best, Move iid_move,
165  Move killer_1, Move killer_2) {
166  // サイド。
167  Side side = engine_ptr_->to_move();
168  Side enemy_side = side ^ 0x3;
169 
170  // 生成開始時のポインタ。
171  std::size_t start = last_;
172 
173  // ナイト、ビショップ、ルーク、クイーンの候補手を作る。
174  for (Piece piece_type = KNIGHT; piece_type <= QUEEN; piece_type++) {
175  Bitboard pieces = engine_ptr_->position()[side][piece_type];
176 
177  for (; pieces; pieces &= pieces - 1) {
178  Square from = Util::GetSquare(pieces);
179 
180  // 各ピースの動き。
181  Bitboard move_bitboard;
182  switch (piece_type) {
183  case KNIGHT:
184  move_bitboard = Util::GetKnightMove(from);
185  break;
186  case BISHOP:
187  move_bitboard = engine_ptr_->GetBishopAttack(from);
188  break;
189  case ROOK:
190  move_bitboard = engine_ptr_->GetRookAttack(from);
191  break;
192  case QUEEN:
193  move_bitboard = engine_ptr_->GetQueenAttack(from);
194  break;
195  default:
196  throw SayuriError("駒の種類が不正です。");
197  break;
198  }
199 
200  // 展開するタイプによって候補手を選り分ける。。
201  if (Type == GenMoveType::NON_CAPTURE) {
202  // キャプチャーじゃない手。
203  move_bitboard &= ~(engine_ptr_->blocker_0());
204  } else {
205  // キャプチャーの手。
206  move_bitboard &= engine_ptr_->side_pieces()[enemy_side];
207  }
208 
209  for (; move_bitboard; move_bitboard &= move_bitboard - 1) {
210  // 手を作る。
211  Move move = 0;
212  move_from(move, from);
213  Square to = Util::GetSquare(move_bitboard);
214  move_to(move, to);
215  move_move_type(move, NORMAL);
216 
217  // ヒストリーの最大値を更新。
218  if (Type == GenMoveType::NON_CAPTURE) {
219  if (engine_ptr_->history()[side][from][to] > history_max_) {
220  history_max_ = engine_ptr_->history()[side][from][to];
221  }
222  }
223 
224  // スタックに登録。
225  move_stack_[last_].move_ = move;
226  last_++;
227  }
228  }
229  }
230 
231  // ポーンの動きを作る。
232  Bitboard pieces = engine_ptr_->position()[side][PAWN];
233  for (; pieces; pieces &= pieces - 1) {
234  Square from = Util::GetSquare(pieces);
235 
236  Bitboard move_bitboard = 0;
237  if (Type == GenMoveType::NON_CAPTURE) {
238  // キャプチャーじゃない手。
239  // ポーンの一歩の動き。
240  move_bitboard = Util::GetPawnMove(from, side)
241  & ~(engine_ptr_->blocker_0());
242  if (move_bitboard) {
243  if (((side == WHITE) && (Util::GetRank(from) == RANK_2))
244  || ((side == BLACK) && (Util::GetRank(from) == RANK_7))) {
245  // ポーンの2歩の動き。
246  move_bitboard |= Util::GetPawn2StepMove(from, side)
247  & ~(engine_ptr_->blocker_0());
248  }
249  }
250  } else {
251  // キャプチャーの手。
252  move_bitboard = Util::GetPawnAttack(from, side)
253  & engine_ptr_->side_pieces()[enemy_side];
254  // アンパッサンがある場合。
255  if (engine_ptr_->en_passant_square()) {
256  move_bitboard |= Util::SQUARE[engine_ptr_->en_passant_square()]
257  & Util::GetPawnAttack(from, side);
258  }
259  }
260 
261  for (; move_bitboard; move_bitboard &= move_bitboard - 1) {
262  // 手を作る。
263  Move move = 0;
264  move_from(move, from);
265  Square to = Util::GetSquare(move_bitboard);
266  move_to(move, to);
267 
268  // ヒストリーの最大値を更新。
269  if (Type == GenMoveType::NON_CAPTURE) {
270  if (engine_ptr_->history()[side][from][to] > history_max_) {
271  history_max_ = engine_ptr_->history()[side][from][to];
272  }
273  }
274 
275  if (engine_ptr_->en_passant_square()
276  && (to == engine_ptr_->en_passant_square())) {
277  move_move_type(move, EN_PASSANT);
278  } else {
279  move_move_type(move, NORMAL);
280  }
281 
282  if (((side == WHITE) && (Util::GetRank(to) == RANK_8))
283  || ((side == BLACK) && (Util::GetRank(to) == RANK_1))) {
284  // 昇格を設定。
285  for (Piece piece_type = KNIGHT; piece_type <= QUEEN; piece_type++) {
286  move_promotion(move, piece_type);
287  move_stack_[last_].move_ = move;
288  last_++;
289  }
290  } else {
291  // 昇格しない場合。
292  move_stack_[last_].move_ = move;
293  last_++;
294  }
295  }
296  }
297 
298  // キングの動きを作る。
299  Square from = engine_ptr_->king()[side];
300  Bitboard move_bitboard = Util::GetKingMove(from);
301  if (Type == GenMoveType::NON_CAPTURE) {
302  // キャプチャーじゃない手。
303  move_bitboard &= ~(engine_ptr_->blocker_0());
304  // キャスリングの動きを追加。
305  if (side == WHITE) {
306  if (engine_ptr_->CanCastling<WHITE_SHORT_CASTLING>()) {
307  move_bitboard |= Util::SQUARE[G1];
308  }
309  if (engine_ptr_->CanCastling<WHITE_LONG_CASTLING>()) {
310  move_bitboard |= Util::SQUARE[C1];
311  }
312  } else {
313  if (engine_ptr_->CanCastling<BLACK_SHORT_CASTLING>()) {
314  move_bitboard |= Util::SQUARE[G8];
315  }
316  if (engine_ptr_->CanCastling<BLACK_LONG_CASTLING>()) {
317  move_bitboard |= Util::SQUARE[C8];
318  }
319  }
320  } else {
321  // キャプチャーの手。
322  move_bitboard &= engine_ptr_->side_pieces()[enemy_side];
323  }
324  for (; move_bitboard; move_bitboard &= move_bitboard - 1) {
325  Move move = 0;
326  move_from(move, from);
327  Square to = Util::GetSquare(move_bitboard);
328  move_to(move, to);
329 
330  // ヒストリーの最大値を更新。
331  if (Type == GenMoveType::NON_CAPTURE) {
332  if (engine_ptr_->history()[side][from][to] > history_max_) {
333  history_max_ = engine_ptr_->history()[side][from][to];
334  }
335  }
336 
337  // キャスリングを設定。
338  if (((side == WHITE) && (from == E1)
339  && ((to == G1) || (to == C1)))
340  || ((side == BLACK) && (from == E8)
341  && ((to == G8) || (to == C8)))) {
342  move_move_type(move, CASTLING);
343  } else {
344  move_move_type(move, NORMAL);
345  }
346 
347  move_stack_[last_].move_ = move;
348  last_++;
349  }
350 
351  ScoreMoves<Type>(start, prev_best, iid_move, killer_1, killer_2, side);
352  }

359  // 候補手に点数をつける。
360  template<GenMoveType Type>
361  void MoveMaker::ScoreMoves(std::size_t start, Move prev_best, Move iid_move,
362  Move killer_1, Move killer_2, Side side) {
363  // 評価値の定義。
364  // 前回の繰り返しでトランスポジションテーブルから得た最善手の点数。
365  constexpr int BEST_MOVE_SCORE = MAX_VALUE;
366  // IIDで得た最善手の点数。
367  constexpr int IID_MOVE_SCORE = BEST_MOVE_SCORE - 1;
368  // 相手キングをチェックする手の点数。
369  constexpr int CHECKING_MOVE_SCORE = IID_MOVE_SCORE - 1;
370  // 駒を取る手の下限値。20 * 20を目安に設定。
371  constexpr int MIN_CAPTURE_SCORE = 403;
372  // キラームーブの点数。
373  constexpr int KILLER_1_MOVE_SCORE = MIN_CAPTURE_SCORE - 1;
374  constexpr int KILLER_2_MOVE_SCORE = KILLER_1_MOVE_SCORE - 1;
375  // ヒストリーの点数の最大値。
376  constexpr std::uint64_t MAX_HISTORY_SCORE = KILLER_2_MOVE_SCORE - 1;
377  // 悪い取る手の点数。
378  constexpr int BAD_CAPTURE_SCORE = -1;
379 
380  Bitboard enemy_king_bb = Util::SQUARE[engine_ptr_->king()[side ^ 0x3]];
381  for (std::size_t i = start; i < last_; i++) {
382  // 手の情報を得る。
383  Square from = move_from(move_stack_[i].move_);
384  Square to = move_to(move_stack_[i].move_);
385 
386  // 相手キングをチェックする手かどうか調べる。
387  bool is_checking_move = false;
388  switch (engine_ptr_->piece_board()[from]) {
389  case PAWN:
390  if ((enemy_king_bb & Util::GetPawnAttack(to, side))) {
391  is_checking_move = true;
392  }
393  break;
394  case KNIGHT:
395  if ((enemy_king_bb & Util::GetKnightMove(to))) {
396  is_checking_move = true;
397  }
398  break;
399  case BISHOP:
400  if ((enemy_king_bb & engine_ptr_->GetBishopAttack(to))) {
401  is_checking_move = true;
402  }
403  break;
404  case ROOK:
405  if ((enemy_king_bb & engine_ptr_->GetRookAttack(to))) {
406  is_checking_move = true;
407  }
408  break;
409  case QUEEN:
410  if ((enemy_king_bb & engine_ptr_->GetQueenAttack(to))) {
411  is_checking_move = true;
412  }
413  break;
414  default:
415  // 何もしない。
416  break;
417  }
418 
419  // 特殊な手の点数をつける。
420  if (EqualMove(move_stack_[i].move_, prev_best)) {
421  // 前回の最善手。
422  move_stack_[i].score_ = BEST_MOVE_SCORE;
423  } else if (EqualMove(move_stack_[i].move_, iid_move)) {
424  // IIDムーブ。
425  move_stack_[i].score_ = IID_MOVE_SCORE;
426  } else if (is_checking_move) {
427  // 相手キングをチェックする手。
428  move_stack_[i].score_ = CHECKING_MOVE_SCORE;
429  } else if (EqualMove(move_stack_[i].move_, killer_1)) {
430  // キラームーブ。
431  move_stack_[i].score_ = KILLER_1_MOVE_SCORE;
432  } else if (EqualMove(move_stack_[i].move_, killer_2)) {
433  // キラームーブ。
434  move_stack_[i].score_ = KILLER_2_MOVE_SCORE;
435  } else {
436  // その他の手を各候補手のタイプに分ける。
437  if ((Type == GenMoveType::CAPTURE)
438  || (move_stack_[i].move_ & PROMOTION_MASK)) {
439  // SEEで点数をつけていく。
440  move_stack_[i].score_ = engine_ptr_->SEE(move_stack_[i].move_);
441  move_stack_[i].score_ = move_stack_[i].score_ >= 0
442  ? move_stack_[i].score_ + MIN_CAPTURE_SCORE : BAD_CAPTURE_SCORE;
443  } else {
444  // ヒストリーを使って点数をつけていく。
445  move_stack_[i].score_ = (engine_ptr_->history()[side][from][to]
446  * MAX_HISTORY_SCORE) / history_max_;
447  }
448  }
449  }
450  }
451  // 実体化。
452  template void MoveMaker::ScoreMoves<GenMoveType::NON_CAPTURE>
453  (std::size_t start, Move best_move, Move iid_move, Move killer_1,
454  Move killer_2, Side side);
455  template void MoveMaker::ScoreMoves<GenMoveType::CAPTURE>
456  (std::size_t start, Move best_move, Move iid_move, Move killer_1,
