102  // テーブルにエントリーを追加する。
103  void TranspositionTable::Add(Hash pos_hash, int depth, int score,
104  ScoreType score_type, Move best_move, int mate_in) {
105  std::unique_lock<std::mutex> lock(mutex_); // ロック。
106 
107  // テーブルのインデックスを得る。
108  std::size_t index = GetTableIndex(pos_hash);
109 
110  // 空いているエントリーへの登録なら使用済みエントリー数をカウント。
111  if (entry_table_[index].depth() <= -MAX_VALUE) {
112  num_used_entries_++;
113  }
114 
115  // テーブルが若い時にに登録されているものなら上書き。
116  // depthがすでに登録されているエントリー以上なら登録。
117  if ((entry_table_[index].table_age() < age_)
118  || (depth >= entry_table_[index].depth())) {
119  entry_table_[index] =
120  TTEntry(pos_hash, depth, score, score_type, best_move, mate_in, age_);
121  }
122  }

124  // 条件を満たすエントリーを返す。
125  const TTEntry& TranspositionTable::GetEntry(Hash pos_hash,
126  int depth) const {
127  // エントリーを得る。
128  std::size_t index = GetTableIndex(pos_hash);
129  if ((entry_table_[index].depth() >= depth)
130  && (entry_table_[index].pos_hash() == pos_hash)) {
131  return entry_table_[index];
132  }
133 
134  // 条件外なので、無効なエントリーを返す。
135  return null_entry_;
136  }
